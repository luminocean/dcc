block := block_unit block
           | NONE

block_unit := statement
           | while_loop
           | if

statement := definition
          | expression ;

definition := type ID = expression

type := BUILTIN_TYPE

expression := expr_unit expr_rest

expr_rest := + expr_unit expr_rest
          | - expr_unit expr_rest
          | * expr_unit expr_rest
          | \ expr_unit expr_rest
          | == expr_uni
          | > expr_unit
          | >= expr_unit
          | < expr_unit
          | <= expr_unit
          | != expr_unit
          | NONE

expr_unit := STRING
           | INTEGER
           | id_expr

id_expr := ID
        | ID = expression

while_loop := WHILE OP expression CP OB block CB

if := IF OP expression CP OB block CB
